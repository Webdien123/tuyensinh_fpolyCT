SELECT A.id, chiso1_1, chiso2_2, ghichu_1, chiso1_2, chiso2_2, ghichu_2, chiso1_3, chiso2_3, ghichu_3 FROM (SELECT diadiem.id, chiso1 AS chiso1_1, chiso2 AS chiso2_1, ghichu AS ghichu_1 FROM diadiem LEFT JOIN tuyensinh on diadiem.id = tuyensinh.id WHERE namhoc = 2016) A LEFT JOIN (SELECT diadiem.id, chiso1 AS chiso1_2, chiso2 AS chiso2_2, ghichu AS ghichu_2 FROM diadiem LEFT JOIN tuyensinh on diadiem.id = tuyensinh.id WHERE namhoc = 2017) B ON A.id = B.id LEFT JOIN (SELECT diadiem.id, chiso1 AS chiso1_3, chiso2 AS chiso2_3, ghichu AS ghichu_3 FROM diadiem LEFT JOIN tuyensinh on diadiem.id = tuyensinh.id WHERE namhoc = 2018) C ON B.id = C.id
